<script>
    let isLoginMode = false;
    const SESSIONS = [
        { id:1, name: 'Detective Noir', date: '21 Feb', time: '13:00h', photogs: 10, models: 6, map: 'https://maps.google.com' },
        { id:2, name: 'Harley Quinn D1', date: '21 Mar', time: '17:00h', photogs: 6, models: 3, map: 'https://maps.google.com' },
        { id:3, name: 'Harley Quinn D2', date: '22 Mar', time: '17:00h', photogs: 6, models: 3, map: 'https://maps.google.com' },
        { id:4, name: 'Pin-Up D1', date: '25 Abr', time: '17:00h', photogs: 10, models: 5, map: 'https://maps.google.com' },
        { id:5, name: 'Pin-Up D2', date: '26 Abr', time: '17:00h', photogs: 10, models: 5, map: 'https://maps.google.com' }
    ];

    function toggleAuthMode() {
        isLoginMode = !isLoginMode;
        document.getElementById('auth-title').innerText = isLoginMode ? 'Bienvenido de nuevo' : 'Únete al Club';
        document.getElementById('reg-fields').style.display = isLoginMode ? 'none' : 'block';
        document.getElementById('toggle-auth').innerText = isLoginMode ? '¿No tienes cuenta? Regístrate aquí' : '¿Ya eres miembro? Inicia sesión aquí';
    }

    function processAuth() {
        const email = document.getElementById('auth-email').value;
        const pass = document.getElementById('auth-pass').value;
        const name = document.getElementById('auth-name').value || 'Félix';

        if(!email || !pass) {
            alert("Maddie dice: ¡Rellena los campos!");
            return;
        }

        // Simulación de Login/Registro
        const user = { email: email, name: name, pts: 0 };
        localStorage.setItem('ms_user', JSON.stringify(user));
        loadApp(user);
    }

    function loadApp(user) {
        document.getElementById('view-auth').classList.add('hidden-view');
        document.getElementById('view-home').classList.remove('hidden-view');
        document.getElementById('navbar').classList.remove('hidden-view');
        document.getElementById('display-name').innerText = user.name;
        
        if(user.email === 'info@bartolomephotography.es') {
            document.getElementById('admin-gear').classList.remove('hidden');
            document.getElementById('admin-indicator').classList.remove('hidden');
        }
        renderSessions();
    }

    function renderSessions() {
        const container = document.getElementById('catalog');
        container.innerHTML = '';
        SESSIONS.forEach(s => {
            container.innerHTML += `
                <div class="glass-card p-5">
                    <div class="flex justify-between items-start mb-2">
                        <h4 class="font-bold text-orange-500 uppercase italic text-sm">${s.name}</h4>
                        <span class="text-[9px] bg-zinc-800 border border-orange-900 px-3 py-1 rounded">15€</span>
                    </div>
                    <p class="text-[10px] text-gray-400 mb-4">${s.date} — Sesión ${s.time}</p>
                    <div class="flex justify-between text-[10px] mb-4 text-gray-500">
                        <span><i class="fas fa-camera mr-1"></i> ${s.photogs} Plazas</span>
                        <span><i class="fas fa-user-friends mr-1"></i> ${s.models} Modelos</span>
                    </div>
                    <div class="flex gap-2">
                        <button onclick="payNotify()" class="neon-btn flex-1 py-3 rounded text-[9px] uppercase tracking-widest">Reservar Plaza</button>
                        <a href="${s.map}" target="_blank" class="bg-zinc-800 p-3 rounded text-zinc-400"><i class="fas fa-map-marked-alt"></i></a>
                    </div>
                </div>`;
        });
    }

    function switchView(id) {
        document.getElementById('view-home').classList.add('hidden-view');
        document.getElementById('view-admin').classList.add('hidden-view');
        document.getElementById(id).classList.remove('hidden-view');
    }

    function payNotify() {
        const code = "MAD-" + Math.floor(1000 + Math.random() * 9000);
        alert(`¡Reserva Iniciada!\n\n1. Envía 15€ a jfelixbart@yahoo.es (PayPal) o Bizum.\n2. Concepto obligatorio: ${code}\n\nFélix confirmará tu plaza en la App tras el pago.`);
    }

    function maddieTip() {
        const tips = ["¡Limpia tu sensor hoy!", "Usa focal fija para Noir", "Harley Quinn pide color", "¡Sube el ISO sin miedo!"];
        document.getElementById('maddie-bubble').innerText = tips[Math.floor(Math.random()*tips.length)];
    }
